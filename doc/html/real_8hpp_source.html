108  void calculate_operation_boundaries() {
109 
110  switch (this->_real_ptr->_operation) {
111 
112  case OPERATION::ADDITION:
113  boost::real::helper::add_boundaries(
114  this->_lhs_it_ptr->approximation_interval.lower_bound,
115  this->_rhs_it_ptr->approximation_interval.lower_bound,
116  this->approximation_interval.lower_bound
117  );
118 
119  boost::real::helper::add_boundaries(
120  this->_lhs_it_ptr->approximation_interval.upper_bound,
121  this->_rhs_it_ptr->approximation_interval.upper_bound,
122  this->approximation_interval.upper_bound
123  );
124  break;
125 
126 
127  case OPERATION::SUBTRACT:
128  boost::real::helper::subtract_boundaries(
129  this->_lhs_it_ptr->approximation_interval.lower_bound,
130  this->_rhs_it_ptr->approximation_interval.upper_bound,
131  this->approximation_interval.lower_bound
132  );
133 
134  boost::real::helper::subtract_boundaries(
135  this->_lhs_it_ptr->approximation_interval.upper_bound,
136  this->_rhs_it_ptr->approximation_interval.lower_bound,
137  this->approximation_interval.upper_bound
138  );
139  break;
140 
141 
142  case OPERATION::MULTIPLICATION: {
143  bool lhs_positive = this->_lhs_it_ptr->approximation_interval.positive();
144  bool rhs_positive = this->_rhs_it_ptr->approximation_interval.positive();
145  bool lhs_negative = this->_lhs_it_ptr->approximation_interval.negative();
146  bool rhs_negative = this->_rhs_it_ptr->approximation_interval.negative();
147 
148  if (lhs_positive && rhs_positive) { // Positive - Positive
149  boost::real::helper::multiply_boundaries(
150  this->_lhs_it_ptr->approximation_interval.lower_bound,
151  this->_rhs_it_ptr->approximation_interval.lower_bound,
152  this->approximation_interval.lower_bound
153  );
154 
155  boost::real::helper::multiply_boundaries(
156  this->_lhs_it_ptr->approximation_interval.upper_bound,
157  this->_rhs_it_ptr->approximation_interval.upper_bound,
158  this->approximation_interval.upper_bound
159  );
160 
161  } else if (lhs_negative && rhs_negative) { // Negative - Negative
162  boost::real::helper::multiply_boundaries(
163  this->_lhs_it_ptr->approximation_interval.upper_bound,
164  this->_rhs_it_ptr->approximation_interval.upper_bound,
165  this->approximation_interval.lower_bound
166  );
167 
168  boost::real::helper::multiply_boundaries(
169  this->_lhs_it_ptr->approximation_interval.lower_bound,
170  this->_rhs_it_ptr->approximation_interval.lower_bound,
171  this->approximation_interval.upper_bound
172  );
173  } else if (lhs_negative && rhs_positive) { // Negative - Positive
174  boost::real::helper::multiply_boundaries(
175  this->_lhs_it_ptr->approximation_interval.lower_bound,
176  this->_rhs_it_ptr->approximation_interval.upper_bound,
177  this->approximation_interval.lower_bound
178  );
179

182  this->_rhs_it_ptr->approximation_interval.lower_bound,

184  );
185 
186  } else if (lhs_positive && rhs_negative) { // Positive - Negative
187  boost::real::helper::multiply_boundaries(
188  this->_lhs_it_ptr->approximation_interval.upper_bound,
189  this->_rhs_it_ptr->approximation_interval.lower_bound,
190  this->approximation_interval.lower_bound
191  );
192 
193  boost::real::helper::multiply_boundaries(
194  this->_lhs_it_ptr->approximation_interval.lower_bound,
195  this->_rhs_it_ptr->approximation_interval.upper_bound,
196  this->approximation_interval.upper_bound
197  );
198 
199  } else { // One is around zero all possible combinations are be tested
200 
201  boundary current_boundary;
202 
203  // Lower * Lower
204  boost::real::helper::multiply_boundaries(
205  this->_lhs_it_ptr->approximation_interval.lower_bound,
206  this->_rhs_it_ptr->approximation_interval.lower_bound,
207  current_boundary
208  );
209 
210  this->approximation_interval.lower_bound = current_boundary;
211  this->approximation_interval.upper_bound = current_boundary;
212 
213  // Upper * upper
214  boost::real::helper::multiply_boundaries(
215  this->_lhs_it_ptr->approximation_interval.upper_bound,
216  this->_rhs_it_ptr->approximation_interval.upper_bound,
217  current_boundary
218  );
219 
220  if (current_boundary < this->approximation_interval.lower_bound) {
221  this->approximation_interval.lower_bound = current_boundary;
222  }
223 
224  if (this->approximation_interval.upper_bound < current_boundary) {
225  this->approximation_interval.upper_bound = current_boundary;
226  }
227 
228  // Lower * upper
229  boost::real::helper::multiply_boundaries(
230  this->_lhs_it_ptr->approximation_interval.lower_bound,
231  this->_rhs_it_ptr->approximation_interval.upper_bound,
232  current_boundary
233  );
234 
235  if (current_boundary < this->approximation_interval.lower_bound) {
236  this->approximation_interval.lower_bound = current_boundary;
237  }
238 
239  if (this->approximation_interval.upper_bound < current_boundary) {
240  this->approximation_interval.upper_bound = current_boundary;
241  }
242 
243  // Upper * lower
244  boost::real::helper::multiply_boundaries(
245  this->_lhs_it_ptr->approximation_interval.upper_bound,
246  this->_rhs_it_ptr->approximation_interval.lower_bound,
247  current_boundary

256  }

516  int max_precision() const {
517 
518  int precision;

523  break;
524 
525  case KIND::ALGORITHM:

527  break;
528 
529  case KIND::OPERATION:

533  precision = std::max(this->_lhs_ptr->max_precision(), this->_rhs_ptr->max_precision());

537  break;

718  bool operator<(const real& other) const {
719  auto this_it = this->cbegin();
720  auto other_it = other.cbegin();
721 
722  int current_precision = std::max(this->max_precision(), other.max_precision());
723  for (int p = 0; p < current_precision; ++p) {
724  // Get more precision
725  ++this_it;
726  ++other_it;
727 
728  if (this_it.approximation_interval < other_it.approximation_interval) {
729  return true;
730  }
731 
732  if (other_it.approximation_interval < this_it.approximation_interval) {
733  return false;
734  }
735  }
736 
737  // If the precision is reached and the number ranges still overlap, then we cannot
738  // know if they are equals or other is less than this and we throw an error.
739  throw boost::real::precision_exception();
740  }

753  bool operator==(const real& other) const {
754  auto this_it = this->cbegin();
755  auto other_it = other.cbegin();
756 
757  int current_precision = std::max(this->max_precision(), other.max_precision());
758  for (int p = 0; p < current_precision; ++p) {
759  // Get more precision
760  ++this_it;
761  ++other_it;
762 
763  bool this_full_precision = this_it.approximation_interval.is_a_number();
764  bool other_full_precision = other_it.approximation_interval.is_a_number();
765  if (this_full_precision && other_full_precision) {
766  return this_it.approximation_interval == other_it.approximation_interval;
767  }
768  }
769 
770  // If the precision is reached and the numbers full precision is not reached, then
771  // we cannot know if they are equals or not.
772  throw boost::real::precision_exception();
773  }
774  };
