113  void iterate_n_times(int n) {
114 
115  // If the number is negative, bounds are interpreted as mirrored:
116  // First, the operation is made as positive, and after bound calculation
117  // bounds are swapped to come back to the negative representation.
118  this->check_and_swap_boundaries();
119 
120  for (int i = 0; i < n; i++) {
121  this->approximation_interval.lower_bound.push_back((*this->_real_ptr)[this->_n + i]);
122  }
123 
124  this->approximation_interval.upper_bound.clear();
125  this->approximation_interval.upper_bound.digits.resize(this->approximation_interval.lower_bound.size());
126  int carry = 1;
127  for (int i = (int)this->approximation_interval.lower_bound.size() - 1; i >= 0; --i) {
128  if (this->approximation_interval.lower_bound[i] + carry == 10) {
129  this->approximation_interval.upper_bound[i] = 0;
130  } else {
131  this->approximation_interval.upper_bound[i] = this->approximation_interval.lower_bound[i] + carry;
132  carry = 0;
133  }
134  }
135 
136  if (carry > 0) {
137  this->approximation_interval.upper_bound.push_front(carry);
138  this->approximation_interval.upper_bound.exponent = this->approximation_interval.lower_bound.exponent + 1;
139  } else {
140  this->approximation_interval.upper_bound.exponent = this->approximation_interval.lower_bound.exponent;
141  }
142 
143  // Left normalization of boundaries representation
144  this->approximation_interval.lower_bound.normalize_left();
145  this->approximation_interval.upper_bound.normalize_left();
146 
147  this->check_and_swap_boundaries();
148  this->_n += n;
149  }
