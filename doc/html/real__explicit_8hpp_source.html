116  void iterate_n_times(int n) {
117  // If the explicit number full precision has been already reached (the end)
118  // is the end of the iterator
119  if (this->_n >= (int)this->_real_ptr->_digits.size()) {
120  // TODO: Remove commented lines that are deprecatod code
121  //this->approximation_interval.lower_bound.push_back(0);
122  //this->approximation_interval.upper_bound.push_back(0);
123  //this->_n++;
124  return;
125  }
126 
127  // If the number is negative, boundaries are interpreted as mirrored:
128  // First, the operation is made as positive, and after boundary calculation
129  // boundaries are swapped to come back to the negative representation.
130  this->check_and_swap_boundaries();
131 
132  // If the explicit number just reaches the full precision (the end)
133  // then set both boundaries are equals
134  if (this->_n + n >= (int)this->_real_ptr->_digits.size()) {
135 
136  for(int i = this->_n; i < (int)this->_real_ptr->_digits.size(); i++) {
137  this->approximation_interval.lower_bound.push_back(this->_real_ptr->_digits[i]);
138  }
139  this->approximation_interval.upper_bound = this->approximation_interval.lower_bound;
140 
141 
142  } else {
143 
144  // If the explicit number didn't reaches the full precision (the end)
145  // then the number interval is defined by truncation.
146 
147  for(int i = 0; i < n; i++) {
148  this->approximation_interval.lower_bound.push_back(this->_real_ptr->_digits[this->_n]);
149  }
150 
151  this->approximation_interval.upper_bound.clear();
152  this->approximation_interval.upper_bound.digits.resize(this->approximation_interval.lower_bound.size());
153 
154  int carry = 1;
155  for (int i = (int)this->approximation_interval.lower_bound.size() - 1; i >= 0; --i) {
156  if (this->approximation_interval.lower_bound[i] + carry == 10) {
157  this->approximation_interval.upper_bound[i] = 0;
158  } else {
159  this->approximation_interval.upper_bound[i] = this->approximation_interval.lower_bound[i] + carry;
160  carry = 0;
161  }
162  }
163 
164  if (carry > 0) {
165  this->approximation_interval.upper_bound.push_front(carry);
166  this->approximation_interval.upper_bound.exponent = this->approximation_interval.lower_bound.exponent + 1;
167  } else {
168  this->approximation_interval.upper_bound.exponent = this->approximation_interval.lower_bound.exponent;
169  }
170  }
171 
172  // Left normalization of boundaries representation
173  this->approximation_interval.lower_bound.normalize_left();
174  this->approximation_interval.upper_bound.normalize_left();
175 
176  this->check_and_swap_boundaries();
177  this->_n = std::min(this->_n + n, (int)this->_real_ptr->_digits.size());
178  }

226  explicit real_explicit(const std::string& number) {
227  // Check that is not an empty string
228  if (number.length() == 0) {
229  throw boost::real::invalid_string_number();
230  }
231 
232  // Check that there is no more that one '.' symbol
233  unsigned int dot_amount = 0;
234  for (const auto& c : number ) {
235  if (c == '.' && dot_amount >= 1) {
236  throw boost::real::invalid_string_number();
237  } else if (c == '.') {
238  dot_amount++;
239  }
240  }
241 
242  int exponent = 0;
243  dot_amount = 0;
244 
245  // Check whether the number is explicitly specified as positive or negative
246  unsigned int first_index = 0;
247  if (number.at(first_index) == '+') {
248  this->_positive = true;
249  first_index++;
250  } else if (number.at(first_index) == '-') {
251  this->_positive = false;
252  first_index++;
253  }
254 
255  // Remove zeros from the lefts side
256  // Note: We know at this point that number.length > 0 because the first check
257  unsigned int last_index = (unsigned int)number.length() - 1;
258  while (last_index > first_index && (number.at(last_index) == '0' || number.at(last_index) == '.')) {
259  if (number.at(last_index) == '.') {
260  dot_amount++;
261  }
262  last_index--;
263  }
264 
265  // The number is all composed by zeros, then it is zero
266  if (last_index == first_index && number.at(last_index) == '0') {
267  this->_digits = {0};
268  this->_exponent = 1;
269  return;
270  }
271 
272  // Remove zeros from the right side
273  // Note: if the number is all made by zeros, then the code never reaches this part
274  // This is why we don't need to check that first_index is lower than the string length
275  while (number.at(first_index) == '0') {
276  first_index++;
277  }
278 
279  if (number.at(first_index) == '.') {
280  dot_amount++;
281  first_index++;
282  while (number.at(first_index) == '0') {
283  first_index++;
284  exponent--;
285  }
286  }
287 
288  for (unsigned int i = first_index; i <= last_index; i++) {
289 
290  if (number.at(i) == '.') {
291  dot_amount++;
292 
293  } else {
294  // This will only affect if the '.' appears, in that case, exponent
295  // wasn't negative and no inconsistency is made.
296  if (dot_amount == 0) exponent++;
297  try {
298  this->_digits.push_back(number.at(i) - '0');
299  } catch (const std::invalid_argument& e) {
300  throw boost::real::invalid_string_number();
301  }
302 
303  }
304  }
305 
306  this->_exponent = exponent;
307  };
